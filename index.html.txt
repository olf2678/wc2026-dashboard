<script>

/* 🔧 CONNECTE TON API ICI */
const API_BASE = "https://wc2026api.com/api"; // à remplacer const API_KEY = "wc26_8QSCAceSkc7V8zifCDG4Bw";

/* MATCHS */
async function loadMatches() {
  try {
    const res = await fetch(`${API_BASE}/matches/today?key=${API_KEY}`);
    const data = await res.json();

    const el = document.getElementById("matches");
    el.innerHTML = "";

    data.matches.forEach(m => {
      el.innerHTML += `
        <div class="match">
          <div>
            <div class="team">${m.home_team}</div>
            <div class="team">${m.away_team}</div>
          </div>

          <div class="score">
            ${m.home_score ?? 0} - ${m.away_score ?? 0}
          </div>
        </div>
      `;
    });

  } catch (e) {
    document.getElementById("matches").innerHTML =
      "Erreur chargement matchs";
  }
}


/* CLASSEMENTS */
async function loadStandings() {
  try {
    const res = await fetch(`${API_BASE}/standings?key=${API_KEY}`);
    const data = await res.json();

    const el = document.getElementById("standings");
    el.innerHTML = "";

    data.groups.forEach(g => {

      let html = `<div class="group">
        <h3>Groupe ${g.name}</h3>`;

      g.teams.forEach(t => {
        html += `
          <div class="row">
            <div>${t.name}</div>
            <div class="points">${t.points} pts</div>
          </div>
        `;
      });

      html += `</div>`;
      el.innerHTML += html;
    });

  } catch (e) {
    document.getElementById("standings").innerHTML =
      "Erreur chargement classements";
  }
}


/* AUTO REFRESH */
async function refresh() {
  await loadMatches();
  await loadStandings();
}

refresh();
setInterval(refresh, 10000);
